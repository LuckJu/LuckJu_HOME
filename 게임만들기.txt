게임 제작


- 바뀐 점
1. 땅을 파서 몬스터를 피해 금을 캐는 스토리가 아닌
   빈 공간에서 몬스터를 피해 별을 먹는 스토리로 교체 하였습니다.

2. 제시된 '.' 길을 그리게 되면 눈이 아파서 공백으로 전환하였습니다.

3. 'M', '@', '#'의 크기가 작고 뚜렷하게 보이거나 구분이 쉽지 않아서 임의의 문자인
   '▲', '★', '●'로 바꾸었습니다.

4. 10 X 10 크기가 너무 작아 20 X 11 ( 테두리 제외 )로 넓혔습니다.

5. 맵에 비해 몬스터가 적어 몬스터의 수를 2에서 3으로 증가시켰습니다.

6. 'q'를 눌러 바로 종료가 아닌 게임 정지후 종료할건지 선택하게 바꾸었습니다.



- 추가한 점
1. 게임 시작 메인화면을 제작하였습니다.

2. 조작법 화면을 따로 하였습니다.

3. 게임을 어떻게 하는지 설명하는 화면을 따로 하였습니다.

4. 별을 먹은 수를 알려주는 카운터를 추가하였습니다.

5. 게임을 정지하는 화면을 따로 하였습니다.

6. 죽었을때 ( 몬스터와 충돌시 ) 따로 나타나는 화면을 제작하였습니다.

7. 몬스터의 움직임은 랜덤으로 상,하,좌,우 1칸 입니다.

	switch(m1){

		case 1:   // 위 

			if(map[y1-1][x1]<COIN){
				map[y1][x1] = 0;
				map[y1-1][x1] = MONSTER1;
			}	break;


		case 2:   // 아래 

			if(map[y1+1][x1]<COIN){
				map[y1][x1] = 0;
				map[y1+1][x1] = MONSTER1;
			}	break;


		case 3:   // 왼쪽 

			if(map[y1][x1-1]<COIN){

			if(map[y1][x1+1]<COIN){
				map[y1][x1] = 0;
				map[y1][x1+1] = MONSTER1;
			}	break;
	}   
}

int MOVE(int x, int y){				//플레이어가 움직이는 함수

	char KEY = _getch();

	switch(KEY){

		case 'i':   // 위 
			if(map[y-1][x]==COIN){
				coin_m++;
				COIN_LOC();
				COIN_CRE();
			}
			
			if(map[y-1][x]<PLAYER){
				map[y][x] = 0;
				map[y-1][x] = PLAYER;
			}else if(map[y-1][x]==MONSTER1 || map[y-1][x]==MONSTER2 || map[y-1][x]==MONSTER3){
				pc+=2;
			}
			
			if(map[y-1][x]==PORTAL){
				clear_check++;
			}
			
			break;


		case 'k':   // 아래 
			if(map[y+1][x]==COIN){
				coin_m++;
				COIN_LOC();
				COIN_CRE();
			}
			
			if(map[y+1][x]<PLAYER){
				map[y][x] = 0;
				map[y+1][x] = PLAYER;
			}else if(map[y+1][x]==MONSTER1 || map[y+1][x]==MONSTER2 || map[y+1][x]==MONSTER3){
				pc+=2;
			}
			
			if(map[y+1][x]==PORTAL){
				clear_check++;
			}
			
			break;


		case 'j':   // 왼쪽 
			if(map[y][x-1]==COIN){
				coin_m++;
				COIN_LOC();
				COIN_CRE();
			}
			
			if(map[y][x-1]<PLAYER){
				map[y][x] = 0;
				map[y][x-1] = PLAYER;
			}else if(map[y][x-1]==MONSTER1 || map[y][x-1]==MONSTER2 || map[y][x-1]==MONSTER3){
				pc+=2;
			}
			
			if(map[y][x-1]==PORTAL){
				clear_check++;
			}
			
			break;


		case 'l':   // 오른쪽 
			if(map[y][x+1]==COIN){
				coin_m++;
				COIN_LOC();
				COIN_CRE();
			}
			
			if(map[y][x+1]<PLAYER){
				map[y][x] = 0;
				map[y][x+1] = PLAYER;
			}else if(map[y][x+1]==MONSTER1 || map[y][x+1]==MONSTER2 || map[y][x+1]==MONSTER3){
				pc+=2;
			}
			
			if(map[y][x+1]==PORTAL){
				clear_check++;
			}
			
			break;

		
		case 'q':   // 종료
			pc++;
			break;
	}
}


void COIN_LOC(){					//코인의 위치 찾는 함수 
	int x,y;

	for( y=0; y<YMAP; y++ ){
		for( x=0; x<XMAP; x++ ){
			if(map[y][x]==COIN){
			xc=x;
			yc=y;
			}
		}
	}
}


void COIN_CRE(){					//코인을 생성하는 함수 
	
	map[yc][xc] = 0;

	int xc = rand() % (XMAP-2) + 1;
	int yc = rand() % (YMAP-2) + 1;
	
	if(map[yc][xc]!=0){
		COIN_CRE();
	}else{
		map[yc][xc] = COIN;
	}
}


void PORTAL_CRE(){					//포탈을 생성하는 함수 

	if(por==0){
		
		por++;
		
		int xpo = rand() % (XMAP-2) + 1;
		int ypo = rand() % (YMAP-2) + 1;
	
		if(map[ypo][xpo]!=0){
			PORTAL_CRE();
		}else{
			map[ypo][xpo] = PORTAL;
		}
	} 
}
